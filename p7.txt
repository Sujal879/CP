#include <stdio.h>

void main()
{
    int a=1;
    printf("a's value is now %d\n",a++);
    printf("a's value is now %d\n",a);
    a=1;
    printf("a's value is now %d\n",++a);
    printf("a's value is now %d\n",a);
    a=1;
    printf("a's value is now %d\n",a--);
    printf("a's value is now %d\n",a);
    a=1;
    printf("a's value is now %d\n",--a);
    printf("a's value is now %d\n",a);




}
