#include <stdio.h>

int main()
{
    int i,j;
    int no1[3][4],no2 [3][4];
    printf("Enter the elements of 3x4 array no1\n");
    for(i=0;i<3;i++)
    for(j=0;j<4;j++)
    scanf("%d", &no1 [i][j]);
    
    printf("Enter the elements of 3x4 array no2\n");
    for(i=0;i<3;i++)
    for(j=0;j<4;j++)
    scanf("%d", &no2 [i][j]);
    
    printf("The 3x4 array no1 is\n");
    for(i=0;i<3;i++)
    {
        for(j=0;j<4;j++)
        printf ("%3d", no1[i][j]);
        printf("\n");
    }
    printf("The 3x4 array no2 is\n");
    for(i=0;i<3;i++)
    {
        for(j=0;j<4;j++)
        printf ("%3d", no2[i][j]);
        printf("\n");
    }
    
    printf("The sum of no1 and no2 is\n");
    for (i=0;i<3;i++)
    {
        for (j=0;j<4;j++)
        printf("%3d",(no1[i][j] + no2[i][j]));
        printf("\n");
    }
    return 0;
}
