#include<stdio.h>
 int swap(int *a,int *b)
 {
     int t;
     t=*a;
     *a=*b;
     *b=t;
 }
 
 int main()
 {
     int i,j;
     printf("Enter the values::");
     scanf("%d%d",&i,&j);
     printf("Before swaping numbers are %d and %d \n",i,j);
     swap(&i,&j);
     printf("After swaping numbers are %d & %d ",i,j);
     return 0;
     
 }  
 
 Output 
Enter the values::6
7
Before swaping numbers are 6 and 7 
After swaping numbers are 7 & 6
